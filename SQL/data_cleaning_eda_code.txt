SELECT *
FROM users;

--- procedo controllando se ci sono duplicati 

SELECT user_id,
COUNT(*) as num_user
FROM users
GROUP BY user_id 
HAVING COUNT(*) > 1;


 --- trovo 3 duplicati, quindi procedo a rimuoverli dal dataset

WITH ranked_user AS (
SELECT *,
ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY user_id) AS rn
FROM users
)
SELECT *
FROM ranked_user
WHERE rn = 1;

--- pulisco la colonna phone_number poichè il formato non è uniforme, utilizzo una CTE per rimuovere i valori non numerici, rimuovo il numero 1 iniziale nel caso in cui ci sono 11 cifre, definisco i valori nulli qualora presenti e rendo il formato del numero di telefono standard 

WITH phones AS (
select phone_number,
REGEXP_REPLACE(phone_number, '[^0-9]', '', 'g') AS digits
FROM customers
)
SELECT phone_number,
CASE
	WHEN LENGTH(digits) = 11 AND LEFT(digits,1) = '1'
    THEN
    CONCAT(SUBSTRING(RIGHT(digits,10),1,3), '-',
           SUBSTRING(RIGHT(digits,10),4,3), '-',
           SUBSTRING(RIGHT(digits,10),7,4)
          )
    WHEN LENGTH(digits) NOT IN (10,11) THEN NULL
ELSE
CONCAT(SUBSTRING(digits,1,3), '-',
       SUBSTRING(digits,4,3), '-',
       SUBSTRING(digits,7,4)
       )
END AS clean_phone_number
FROM phones;

--- mentre qui creo il nuovo dataset customers pulito 

CREATE TABLE clean_customers AS
WITH ranked_cust AS (
SELECT *,
ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY customer_id) AS rn
FROM customers
),
phones_cte AS (
SELECT *,
REGEXP_REPLACE(phone_number,'[^0-9]','','g') AS digits
FROM ranked_cust
)
SELECT customer_id,
user_id,
INITCAP(name) AS clean_name,
age,
gender,
CASE 
	WHEN email NOT LIKE '%@%' THEN NULL
ELSE LOWER(email)
END AS clean_email,
CASE
	WHEN LENGTH(digits) = 11
	AND LEFT(digits,1) = '1'
	THEN CONCAT(SUBSTRING(RIGHT(digits,10),1,3), '-',
                SUBSTRING(RIGHT(digits,10),4,3), '-',
                SUBSTRING(RIGHT(digits,10),7,4)
               )
	WHEN LENGTH(digits) NOT IN (10,11)
    THEN NULL
ELSE CONCAT(SUBSTRING(digits,1,3), '-',
            SUBSTRING(digits,4,3), '-',
            SUBSTRING(digits,7,4)
            )
END AS clean_phone_number,
INITCAP(city) AS clean_city,
CASE
	WHEN state IN ('Texas', 'tx') THEN 'TX'
	WHEN state IN ('ca', 'Calif.') THEN 'CA'
	WHEN state = 'nv' THEN 'NV'
	WHEN state = 'nc' THEN 'NC'
	WHEN state = 'co' THEN 'CO'
	WHEN state = 'ma' THEN 'MA'
	WHEN state = 'ny' THEN 'NY'
	WHEN state = 'oh' THEN 'OH'
	WHEN state = 'pa' THEN 'PA'
	WHEN state = 'tn' THEN 'TN'
	WHEN state = 'wa' THEN 'WA'
ELSE state
END AS clean_state,
signup_date,
preferred_contact_method
FROM phones_cte
WHERE rn = 1;
